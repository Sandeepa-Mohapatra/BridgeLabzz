        public void deletAt(int index)
        {
            if(index==0)
            {
                head = head.next;
            }
            else
            {
                Node n = head;
                Node n1 = null;
                for(int i=0;i<index-1;i++)

                    n = n.next;
                }
                n1 = n.next;
                n.next = n1.next;
            }
        }

        public int[] BubbleSortI(int[] a)
        {
            int temp;
            for (int j = 0; j <= a.Length - 2; j++)
            {
                for (int i = 0; i <= a.Length - 2; i++)
                {
                    if (a[i] > a[i + 1])
                    {
                        temp = a[i + 1];
                        a[i + 1] = a[i];
                        a[i] = temp;
                    }
                }
            }
            return a;
        }
